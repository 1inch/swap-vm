\begin{abstract}
This document explains the \texttt{XYCConcentrate} GrowLiquidity mechanism
used in the 1inch SwapVM, starting from first principles with worked numerical
examples.
We then identify two critical bugs found during the Bail Security audit---swapped
delta formulas and a fundamentally broken 3D/4D generalization---and show
exactly how to fix each one.
Finally, we analyze the additivity (path-independence) properties under
different fee configurations.
\end{abstract}

\part{How the Mechanism Works}

\section{The Idea: Concentrated Liquidity on a Constant-Product Curve}

A plain constant-product AMM holds two tokens with reserves $(x, y)$ satisfying
$xy = k$.  Liquidity is spread across all prices from~$0$ to~$\infty$,
which is capital-inefficient.

\textbf{Concentrated liquidity} restricts trading to a chosen price range
$[P_{\mathrm{lo}},\, P_{\mathrm{hi}}]$.
The trick is to add \emph{virtual offsets} (``deltas'') to the real balances
so that the constant-product formula acts on larger \emph{virtual reserves}:
\[
X = b_x + \delta_x, \qquad Y = b_y + \delta_y,
\]
where $b_x, b_y$ are the real token balances and $\delta_x, \delta_y$ are the
offsets.  Swaps use the virtual reserves $(X, Y)$, but the pool only actually
holds $(b_x, b_y)$.

\begin{example}[Why this helps]\label{ex:why-concentrate}
Suppose a maker deploys $b_x = b_y = 1{,}000$ tokens at price~$1$
and concentrates between $P_{\mathrm{lo}} = 0.5$ and $P_{\mathrm{hi}} = 2.0$.

Without concentration the pool would be $(1000, 1000)$ with $k = 10^6$.

With concentration the virtual reserves are much larger (say $X = Y = 3{,}000$),
giving $k = 9 \times 10^6$---nine times the effective liquidity.
The trade $\Delta = 100$ of token~$X$ yields:
\begin{itemize}
\item Without concentration:
  $\mathrm{out} = 100 \times 1000 / 1100 \approx 90.9$.
\item With concentration:
  $\mathrm{out} = 100 \times 3000 / 3100 \approx 96.8$.
\end{itemize}
The concentrated pool gives a better price because liquidity is deeper around
the current price.
\end{example}


\section{The Standard Math (Uniswap V3 Style)}\label{sec:standard-math}

Define the \emph{spot price} $P = Y/X$ and \emph{liquidity}
$L = \sqrt{XY}$.  Then:

\begin{align}
X &= \frac{L}{\sqrt{P}}, &  Y &= L\sqrt{P},  \label{eq:virtual} \\[4pt]
\delta_x &= \frac{L}{\sqrt{P_{\mathrm{hi}}}}, &
\delta_y &= L\sqrt{P_{\mathrm{lo}}},  \label{eq:offsets} \\[4pt]
b_x &= X - \delta_x = L\!\left(\frac{1}{\sqrt{P}}
      - \frac{1}{\sqrt{P_{\mathrm{hi}}}}\right), &
b_y &= Y - \delta_y = L\!\left(\sqrt{P}
      - \sqrt{P_{\mathrm{lo}}}\right).  \label{eq:real}
\end{align}

\begin{remark}[Boundary conditions]
At $P = P_{\mathrm{lo}}$: $b_y = 0$ (all capital in token~$X$).
At $P = P_{\mathrm{hi}}$: $b_x = 0$ (all capital in token~$Y$).
\end{remark}

\begin{example}[Computing deltas from scratch]\label{ex:deltas-from-scratch}
Take $P_{\mathrm{lo}} = 0.25$, $P_{\mathrm{hi}} = 4.0$, $P = 1.0$, $L = 1000$.
\begin{align*}
X &= 1000 / \sqrt{1} = 1000, & Y &= 1000 \times \sqrt{1} = 1000,\\
\delta_x &= 1000 / \sqrt{4} = 500, & \delta_y &= 1000 \times \sqrt{0.25} = 500,\\
b_x &= 1000 - 500 = 500, & b_y &= 1000 - 500 = 500.
\end{align*}
So a maker depositing $b_x = b_y = 500$ real tokens gets virtual reserves of
$(1000, 1000)$---double the effective liquidity.
\end{example}


\section{How \texttt{XYCConcentrate} Works in Practice}\label{sec:impl-overview}

The implementation stores three pieces of data per token pair:
$\delta_x,\; \delta_y,\; L_0$ (the initial liquidity).
On every swap it runs these steps:

\begin{enumerate}[label=\textbf{Step \arabic*},leftmargin=3em]
\item \textbf{Read the current liquidity $L_{\mathrm{cur}}$.}
  For 2D this is $L_{\mathrm{cur}} = \sqrt{K_{\mathrm{stored}}}$, where $K$ was
  saved from the previous swap.

\item \textbf{Build concentrated balances.}
  Scale the deltas proportionally to how much $L$ has grown:
  \begin{equation}\label{eq:concentrate}
  X = b_x + \delta_x \cdot \frac{L_{\mathrm{cur}}}{L_0}, \qquad
  Y = b_y + \delta_y \cdot \frac{L_{\mathrm{cur}}}{L_0}.
  \end{equation}

\item \textbf{Execute a constant-product swap} on $(X, Y)$:
  \begin{equation}\label{eq:swap}
  \mathrm{amtOut} = \frac{\Delta_{\mathrm{eff}} \cdot Y}{X + \Delta_{\mathrm{eff}}},
  \end{equation}
  where $\Delta_{\mathrm{eff}}$ is the input after any fee.

\item \textbf{Update and store $L$.}
  \begin{equation}\label{eq:L-update}
  K_{\mathrm{new}} = (X + \mathrm{amtIn}) \cdot (Y - \mathrm{amtOut}),
  \quad
  L_{\mathrm{new}} = \sqrt{K_{\mathrm{new}}}.
  \end{equation}
\end{enumerate}

\begin{example}[A full swap walkthrough]\label{ex:walkthrough}
Start with $b_x = b_y = 500$, $\delta_x = \delta_y = 500$,
$L_0 = 1000$, and $L_{\mathrm{cur}} = L_0 = 1000$ (first swap, no growth yet).
No fee ($f = 0$).  Trader sends $\Delta = 100$ of token~$X$.

\textbf{Step 2.}
$X = 500 + 500 \times 1000/1000 = 1000$, \;
$Y = 500 + 500 \times 1000/1000 = 1000$.

\textbf{Step 3.}
$\mathrm{amtOut} = 100 \times 1000 / 1100 \approx 90.91$.

\textbf{Step 4.}
$K = 1100 \times 909.09 = 1{,}000{,}000$ (unchanged---no fee), \;
$L = 1000$ (unchanged).

Trader receives $\approx 90.91$ of token~$Y$.
\end{example}

\begin{example}[Swap with input-fee]\label{ex:walkthrough-fee}
Same pool as above.  Fee $f = 1\%$.  Trader sends $\Delta = 100$.

\textbf{Step 2.}  Same: $(X, Y) = (1000, 1000)$.

\textbf{Step 3.}
Effective input $\Delta_{\mathrm{eff}} = 0.99 \times 100 = 99$.
$\mathrm{amtOut} = 99 \times 1000 / 1099 \approx 90.08$.

\textbf{Step 4.}
$X' = 1000 + 100 = 1100$ (full $\Delta$ credited).
$Y' = 1000 - 90.08 = 909.92$.
$K = 1100 \times 909.92 = 1{,}000{,}912 > 10^6$.
$L = \sqrt{1{,}000{,}912} \approx 1000.46$.

The product \emph{grew} because the fee ($1$ token) stayed in the pool.
Next swap, the deltas scale up by $1000.46/1000 = 1.00046$, making the pool
slightly deeper.
\end{example}


\section{How \texttt{computeDeltas} Constructs the Pool}\label{sec:computeDeltas}

The function \texttt{computeDeltas(balanceA, balanceB, price, priceMin, priceMax)}
computes the initial deltas and liquidity.  Its formulas are:
\begin{align}
\texttt{sqrtMin} &= \sqrt{\frac{P}{P_{\mathrm{lo}}}}, &
\texttt{sqrtMax} &= \sqrt{\frac{P_{\mathrm{hi}}}{P}}, \label{eq:impl-sqrts}\\[4pt]
\texttt{deltaA} &= \frac{b_x}{\texttt{sqrtMin} - 1}, &
\texttt{deltaB} &= \frac{b_y}{\texttt{sqrtMax} - 1}, \label{eq:impl-deltas}\\[4pt]
\texttt{liquidity} &= \sqrt{(b_x + \texttt{deltaA})(b_y + \texttt{deltaB})}.
\label{eq:impl-L}
\end{align}

\begin{example}[Running \texttt{computeDeltas}]\label{ex:computeDeltas}
Inputs: $b_x = b_y = 1000$, $P = 1.0$, $P_{\mathrm{lo}} = 0.25$,
$P_{\mathrm{hi}} = 4.0$.
\begin{align*}
\texttt{sqrtMin} &= \sqrt{1.0/0.25} = 2.0, &
\texttt{sqrtMax} &= \sqrt{4.0/1.0} = 2.0,\\
\texttt{deltaA} &= 1000 / (2 - 1) = 1000, &
\texttt{deltaB} &= 1000 / (2 - 1) = 1000,\\
\texttt{liquidity} &= \sqrt{2000 \times 2000} = 2000.
\end{align*}
Virtual reserves: $(X, Y) = (2000, 2000)$.
\end{example}

\begin{example}[Away from the geometric mean]\label{ex:computeDeltas-off}
Same bounds, but now $P = 0.5$ (not the geometric mean $\sqrt{0.25 \times 4} = 1$):
\begin{align*}
\texttt{sqrtMin} &= \sqrt{0.5/0.25} = \sqrt{2} \approx 1.414, &
\texttt{sqrtMax} &= \sqrt{4.0/0.5} = \sqrt{8} \approx 2.828,\\
\texttt{deltaA} &= 1000 / 0.414 \approx 2414, &
\texttt{deltaB} &= 1000 / 1.828 \approx 547.
\end{align*}
We will see in \S\ref{sec:bug-deltas} that these values are \emph{wrong}---the
formulas for A and B are swapped.
\end{example}

\section{Bug 1: The Delta Formulas Are Swapped}\label{sec:bug-deltas}

\noindent\textit{Audit references:}
\begin{itemize}[nosep,leftmargin=2em]
\item \textbf{[H]} \textit{computeDeltas does not obey price limits}
\item \textbf{[M]} \textit{computeDeltas --- ratio of square roots of prices is not meaningful}
\item \textbf{[H]} \textit{computeDeltas --- inconsistent rounding direction}
\end{itemize}

\subsection{The correct formulas}

From the standard math (\S\ref{sec:standard-math}), expressing the offsets
in terms of real balances:
\begin{align}
\delta_x &= \frac{b_x}{\sqrt{P_{\mathrm{hi}}/P} - 1}
\quad\bigl(= b_x\,\sqrt{P}\,/\,(\sqrt{P_{\mathrm{hi}}} - \sqrt{P})\bigr),
\label{eq:delta-x-correct} \\
\delta_y &= \frac{b_y}{\sqrt{P/P_{\mathrm{lo}}} - 1}
\quad\bigl(= b_y\,\sqrt{P_{\mathrm{lo}}}\,/\,(\sqrt{P} - \sqrt{P_{\mathrm{lo}}})\bigr).
\label{eq:delta-y-correct}
\end{align}

\subsection{What the implementation does}

\begin{center}
\renewcommand{\arraystretch}{1.4}
\begin{tabular}{@{}lcc@{}}
\toprule
& \textbf{Correct} & \textbf{Implementation} \\
\midrule
$\delta_x$ (tokenA) &
  $\dfrac{b_x}{\sqrt{P_{\mathrm{hi}}/P} - 1}$ &
  $\dfrac{b_x}{\sqrt{P/P_{\mathrm{lo}}} - 1}$ \\[10pt]
$\delta_y$ (tokenB) &
  $\dfrac{b_y}{\sqrt{P/P_{\mathrm{lo}}} - 1}$ &
  $\dfrac{b_y}{\sqrt{P_{\mathrm{hi}}/P} - 1}$ \\
\bottomrule
\end{tabular}
\end{center}

\bug{The sqrt ratio arguments are swapped between tokenA and tokenB.}

\subsection{When the bug is invisible}

The two formulas coincide when $\sqrt{P_{\mathrm{hi}}/P} = \sqrt{P/P_{\mathrm{lo}}}$,
i.e.\ $P = \sqrt{P_{\mathrm{lo}} \cdot P_{\mathrm{hi}}}$ (the geometric mean of the
bounds).  This is why tests at symmetric prices may not catch the bug.

\subsection{Numerical comparison}

\begin{example}[Correct vs.\ implementation at $P = 0.5$]\label{ex:bug-deltas}
$b_x = b_y = 1000$, $P_{\mathrm{lo}} = 0.25$, $P_{\mathrm{hi}} = 4.0$, $P = 0.5$.

\medskip
\textbf{Correct:}
\[
\delta_x = \frac{1000}{\sqrt{4/0.5} - 1}
        = \frac{1000}{\sqrt{8} - 1}
        = \frac{1000}{1.828}
        \approx 547,
\qquad
\delta_y = \frac{1000}{\sqrt{0.5/0.25} - 1}
        = \frac{1000}{\sqrt{2} - 1}
        = \frac{1000}{0.414}
        \approx 2414.
\]
Virtual reserves: $X = 1547$, $Y = 3414$.
Check: $Y/X = 3414/1547 \approx 2.21$, and $\sqrt{P} = \sqrt{0.5} \approx 0.707$;
$L = \sqrt{XY} = \sqrt{5{,}281{,}158} \approx 2298$.
Verify $b_y$: $L(\sqrt{P} - \sqrt{P_{\mathrm{lo}}}) = 2298 \times (0.707 - 0.5) = 2298 \times 0.207 \approx 476$.

Hmm, this doesn't match $b_y = 1000$. That is because at $P = 0.5$ with these bounds and $b_x = b_y = 1000$, the system is over-constrained---$P$ is not a free parameter.

\medskip
\textbf{Implementation (swapped):}
$\texttt{deltaA} = 2414$, $\texttt{deltaB} = 547$.

The deltas are \emph{exactly exchanged}, giving virtual reserves
$X = 3414$, $Y = 1547$---mirror image of the correct values.
\end{example}

\subsection{The fix}

\fix{Swap the two sqrt expressions in \texttt{computeDeltas}:}
\begin{align*}
\texttt{sqrtForA} &= \sqrt{P_{\mathrm{hi}} / P} &
  &\text{(was: $\sqrt{P / P_{\mathrm{lo}}}$)}, \\
\texttt{sqrtForB} &= \sqrt{P / P_{\mathrm{lo}}} &
  &\text{(was: $\sqrt{P_{\mathrm{hi}} / P}$)}.
\end{align*}


\section{Design Note: \texttt{computeDeltas} Takes Price as Input}\label{sec:overconstrained}

\noindent\textit{Audit reference:}
\begin{itemize}[nosep,leftmargin=2em]
\item \textbf{[H]} \textit{computeDeltas is overspecified}
\end{itemize}

\medskip
\texttt{computeDeltas} takes \textbf{5 inputs}: $b_x, b_y, P, P_{\mathrm{lo}}, P_{\mathrm{hi}}$.
In pure concentrated-liquidity math, only 4 are independent---given
$(b_x, b_y, P_{\mathrm{lo}}, P_{\mathrm{hi}})$ the spot price $P$ is uniquely
determined.

However, this is \textbf{not a bug}.
The maker calls \texttt{computeDeltas} off-chain to construct their own
position.  They control all 5 inputs and deliberately pass the price they
believe reflects the current market.  The function is a \texttt{pure} helper:
no one else can call it with a ``wrong'' $P$ to harm the maker.

If the maker's balances don't perfectly match the claimed price at those bounds,
the resulting deltas produce slightly different concentration characteristics
than the textbook formulas would imply.  This is a capital-efficiency trade-off
the maker accepts, not a security vulnerability.

\begin{remark}[Deployment-timing risk]
There is a real operational risk: the market price can move between when the
maker calls \texttt{computeDeltas} off-chain and when the strategy goes live
and the first swap executes on-chain.
If the price has moved, the pool opens at a stale price.  Arbitrageurs will
trade against the pool until it converges to the market price, extracting
value from the maker.

Concentrated liquidity \emph{amplifies} this effect: higher capital efficiency
means more loss per unit of price movement compared to a full-range pool.

This is an inherent deployment-timing risk for any concentrated-liquidity AMM
(including Uniswap~V3 positions).  It is not specific to the 5-parameter
interface, since deriving $P$ from $(b_x, b_y, P_{\mathrm{lo}}, P_{\mathrm{hi}})$
would still bake in a price that becomes stale after deployment.
The mitigation is for makers to deploy when they can execute quickly
and to choose wider ranges if they expect delays.
\end{remark}

\begin{remark}[Possible improvement]
The function could also return the \emph{implied} $P$ from the quadratic
\[
b_x \sqrt{P_{\mathrm{hi}}}\,(\sqrt{P})^2
  - \bigl(b_y - b_x\sqrt{P_{\mathrm{lo}} P_{\mathrm{hi}}}\bigr)\sqrt{P}
  - b_y\sqrt{P_{\mathrm{lo}}} = 0,
\]
so the maker can compare it against the price they passed and gauge
how much the position diverges from the textbook formulation.
\end{remark}


\section{Note on Stored $L$ (2D Is Fine; $>$2D Is Not)}\label{sec:stored-L}

\noindent\textit{Audit references:}
\begin{itemize}[nosep,leftmargin=2em]
\item \textbf{[H]} \textit{GrowLiquidityXD grows liquidity for unrelated tokens}
\item \textbf{[H]} \textit{Liquidity calculation is incorrect in Fee $\to$ XYCConcentrate sequence}
\item \textbf{[H]} \textit{Liquidity will be disrupted if combined with Decay}
\item \textbf{[H]} \textit{XYCConcentrate $\to$ Fee does not work for protocol fee scenario}
\end{itemize}

\medskip
The current code stores $K = X' Y'$ after each swap and derives $L = \sqrt{K}$
for the next swap.

\textbf{For 2D this is correct.}
The strategy is immutable (the maker cannot deposit or withdraw between swaps)
and Aqua tracks the real balances.  Because no external changes can occur
between swaps, the stored $K$ faithfully reflects the post-swap state and
correctly drives the delta scaling on the next swap.
With input fees $K$ grows monotonically, and the stored value captures
that growth accurately.

\textbf{For $>$2 tokens, storing $L$ is still problematic}
(but this is subsumed by Bug~2 below).
A swap A$\to$B changes $b_B$, which affects the B$\leftrightarrow$C implied
price and $L$.
The pairwise decomposition stores a separate $L$ per pair, so the A$\to$B
swap does not update the B$\leftrightarrow$C state.
This is one more reason the 3D/4D approach is unsound.


\section{Bug 2: 3D/4D Pairwise Decomposition Is Fundamentally Broken}
\label{sec:bug-3d}

\noindent\textit{Audit references:}
\begin{itemize}[nosep,leftmargin=2em]
\item \textbf{[H]} \textit{GrowLiquidityXD decomposition into pairwise AMMs is unsound for $>$2 assets}
\item \textbf{[H]} \textit{GrowLiquidityXD prices don't adjust for uninvolved tokens (self-arbitrage)}
\item \textbf{[H]} \textit{GrowLiquidityXD grows liquidity for unrelated tokens}
\item \textbf{[M]} \textit{computeDeltas is not meaningful for $>$2 assets}
\end{itemize}

\medskip
The 3D and 4D variants treat each token pair as an independent 2D pool.
This is \textbf{unsound}.

\begin{theorem}[Impossibility of Pairwise Decomposition]\label{thm:no-pairwise}
For $n \ge 3$ tokens, a concentrated-liquidity constant-product AMM cannot be
decomposed into $\binom{n}{2}$ independent pairwise AMMs while preserving
price consistency.
\end{theorem}

\begin{proof}[Proof sketch]
For three tokens $A$, $B$, $C$, price consistency requires
$P_{A/C} = P_{A/B} \cdot P_{B/C}$.
A swap $A \to B$ changes $P_{A/B}$ but leaves the independent $B/C$ and $A/C$
pools untouched, so $P_{B/C}$ stays fixed.  Now
$P_{A/B}' \cdot P_{B/C} \ne P_{A/C}$ (which hasn't changed either).
This violates the consistency condition and creates triangular arbitrage:
$A \to B \to C \to A$ can extract value, and repeating it drains the pool.
\end{proof}

\begin{example}[Triangular arbitrage (simplified)]\label{ex:tri-arb}
Three tokens, all initially priced $1{:}1$.
\begin{enumerate}
\item Swap a large amount of $A$ for $B$.  Now $P_{A/B}$ drops (A cheaper, B expensive).
\item The $B/C$ pool still thinks $B$ is worth $1$ unit of $C$.
  Swap $B \to C$ at the stale (too-high) price for $B$.
\item The $A/C$ pool still thinks $A$ is worth $1$ unit of $C$.
  Swap $C \to A$ at the stale price.
\item The attacker ends up with more $A$ than they started with.
\end{enumerate}
Repeat until the pool is drained.
\end{example}

\begin{remark}[PoC exploit]
The Bail Security audit includes a Foundry invariant test that finds
draining swap sequences on a 3-token XYCConcentrate pool.
\end{remark}

\subsection{The fix}

\fix{Disable the 3D and 4D variants.}
A correct $n$-asset concentrated-liquidity CFAMM requires:
\begin{itemize}
\item A single invariant over all $n$ virtual balances (not pairwise).
\item Recomputing $L$ on each swap via $\sum_{i} \ln(p_i + b_i/L) = 0$
  (no closed form for $n \ge 3$; requires Newton iteration).
\item Careful rounding in the direction disadvantageous to the taker.
\end{itemize}
This is substantially more complex than the current approach.

\section{Additivity: What It Means and Why It Matters}\label{sec:additivity-intro}

\begin{definition}[Additivity Types]\label{def:additivity}
Let $\Out(\Delta; s)$ be the trader output for input $\Delta$ from state~$s$,
and let $F_\Delta(s)$ be the post-swap state.
Define $\Out_{\mathrm{split}}(a,b; s) := \Out(a;s) + \Out(b; F_a(s))$.
\begin{enumerate}[label=(\alph*)]
\item \textbf{Strictly additive:} $\Out(a+b;s) = \Out_{\mathrm{split}}(a,b;s)$ for all $a,b\ge 0$.
\item \textbf{Subadditive:} $\Out(a+b;s) > \Out_{\mathrm{split}}(a,b;s)$ for all $a,b > 0$.
\item \textbf{Superadditive:} $\Out(a+b;s) < \Out_{\mathrm{split}}(a,b;s)$ for all $a,b > 0$.
\end{enumerate}
\end{definition}

Why this matters for a DEX:
\begin{itemize}
\item \textbf{Subadditive = safe.}  One-shot is better for the trader.
  Splitting hurts. No MEV exploit from chunking.
\item \textbf{Superadditive = dangerous.}  Splitting is better.
  MEV bots chunk trades to extract value.
\item \textbf{Strictly additive = ideal.}  Path-independent.
  No gaming in either direction.
\end{itemize}

\begin{remark}
The additivity type depends on the \emph{fee placement}, not on the delta
computation.  So the delta bugs from Part~II do not change these results.
\end{remark}


\section{Fee-Free: Strictly Additive}\label{sec:nofee}

\begin{theorem}\label{thm:strict-nofee}
Without fees, the 2D GrowLiquidity mechanism is strictly additive.
\end{theorem}

\begin{proof}
Fix virtual reserves $(X,Y)$.  The fee-free swap preserves $K = XY$:
\[
(X + \Delta)(Y - \mathrm{out}) = XY.
\]
\textbf{One-shot} $\Delta = a+b$:
$\mathrm{out}_{\text{one}} = (a+b)Y / (X+a+b)$.

\textbf{Split} $a$ then $b$:
\begin{align*}
\mathrm{out}_1 &= \frac{aY}{X+a}, &
X_1 &= X+a, & Y_1 &= \frac{XY}{X+a}.\\
\mathrm{out}_2 &= \frac{bY_1}{X_1+b} = \frac{bXY}{(X+a)(X+a+b)}.
\end{align*}
Sum:
\[
\mathrm{out}_1 + \mathrm{out}_2
= \frac{aY}{X+a} + \frac{bXY}{(X+a)(X+a+b)}
= \frac{(a+b)Y}{X+a+b}
= \mathrm{out}_{\text{one}}.
\]
Since $K$ is unchanged after each leg, $L$ doesn't change, and deltas scale
identically, confirming strict additivity.
\end{proof}

\begin{example}\label{ex:strict-num}
$(X,Y) = (1000, 1000)$.

One-shot $200$: $\mathrm{out} = 200{,}000 / 1200 \approx 166.67$.

Split $100 + 100$:
$\mathrm{out}_1 = 100{,}000/1100 \approx 90.91$;
state $(1100, 909.09)$;
$\mathrm{out}_2 = 90{,}909/1200 \approx 75.76$;
total $\approx 166.67$.  Equal. \checkmark
\end{example}


\section{Input-Fee: Subadditive (Safe)}\label{sec:input-fee}

\begin{definition}\label{def:input-fee}
Fee rate $f \in (0,1)$.  Effective input $\Delta_{\mathrm{eff}} = (1-f)\Delta$.
Full $\Delta$ is credited to the input reserve; only $\Delta_{\mathrm{eff}}$ enters
the swap.
\end{definition}

\begin{theorem}\label{thm:sub}
With input-fee, GrowLiquidity is subadditive:
$\Out(a+b) > \Out(a) + \Out(b; F_a)$ for $a,b > 0$.
\end{theorem}

\begin{proof}
\textbf{One-shot:}
$\Out_{\text{one}} = (1-f)(a+b)Y / [X + (1-f)(a+b)]$.

\textbf{Split:}
$\mathrm{out}_1 = (1-f)aY / [X + (1-f)a]$.
Post-swap: $X_1 = X + a$, $Y_1 = XY / [X + (1-f)a]$.
Since $X + a > X + (1-f)a$, the product $K_1 = X_1 Y_1 > XY$ grows.
$\mathrm{out}_2 = (1-f)b \cdot Y_1 / [X_1 + (1-f)b]$.

The key: the second chunk's input reserve is $X_1 = X + a$, but
if there were no fee it would be $X + (1-f)a$.
The extra $fa$ makes the pool look richer in~$X$, giving a worse rate for
the $X \to Y$ direction.
Therefore $\mathrm{out}_1 + \mathrm{out}_2 < \Out_{\text{one}}$.
\end{proof}

\begin{example}[$f = 1\%$]\label{ex:sub-num}
$(X,Y) = (1000, 1000)$, $f = 0.01$.

One-shot $200$: eff.\ $198$; $\mathrm{out} = 198{,}000/1198 \approx 165.28$.

Split $100 + 100$:
Eff.\ $99$; $\mathrm{out}_1 = 99{,}000/1099 \approx 90.08$.
$X_1 = 1100$, $Y_1 = 1{,}000{,}000/1099 \approx 909.92$.
Eff.\ $99$; $\mathrm{out}_2 = 99 \times 909.92/1199 \approx 75.10$.
Total $\approx 165.18$.

One-shot $165.28 > $ split $165.18$: subadditive (safe). \checkmark
\end{example}


\section{Output-Fee: Superadditive (Exploitable)}\label{sec:output-fee}

\begin{definition}\label{def:output-fee}
Fee rate $f \in (0,1)$.  Trader receives $(1-f) \cdot \mathrm{out}_{\mathrm{nf}}$.
The fee $f \cdot \mathrm{out}_{\mathrm{nf}}$ stays in the output reserve.
\end{definition}

\begin{theorem}\label{thm:super}
With output-fee, GrowLiquidity is superadditive:
$\Out(a+b) < \Out(a) + \Out(b; F_a)$ for $a,b > 0$.

The exact gap is:
\begin{equation}\label{eq:super-gap}
\boxed{\mathrm{Gap} = \frac{(1-f)\,f\,ab\,Y}{(X+a)(X+a+b)}.}
\end{equation}
\end{theorem}

\begin{proof}
\textbf{One-shot:}
$\Out_{\text{one}} = (1-f)(a+b)Y / (X+a+b)$.

\textbf{Split:}
$\mathrm{out}_1 = (1-f)aY/(X+a)$.
$Y_1 = Y(X+fa)/(X+a)$ (fee keeps extra $Y$ in pool).
$\mathrm{out}_2 = (1-f)bY(X+fa)/[(X+a)(X+a+b)]$.

\begin{align*}
\mathrm{out}_1 + \mathrm{out}_2
&= \frac{(1-f)Y}{(X+a)(X+a+b)}\bigl[(a+b)(X+a) + fab\bigr],\\[4pt]
\Out_{\text{one}}
&= \frac{(1-f)Y}{(X+a)(X+a+b)}\bigl[(a+b)(X+a)\bigr].
\end{align*}
Difference: $(1-f)f \cdot ab \cdot Y / [(X+a)(X+a+b)] > 0$.
\end{proof}

\begin{example}[$f = 1\%$]\label{ex:super-num}
$(X,Y) = (1000, 1000)$, $f = 0.01$.

One-shot $200$: fee-free out $166.67$; trader gets $0.99 \times 166.67 \approx 165.00$.

Split $100 + 100$:
$\mathrm{out}_1 = 0.99 \times 90.91 \approx 90.00$.
$Y_1 = 1000 \times (1000 + 10)/1100 = 1000 \times 1010/1100 \approx 918.18$.
$\mathrm{out}_2 = 0.99 \times 100 \times 918.18/1200 \approx 75.75$.
Total $\approx 165.75$.

Split $165.75 >$ one-shot $165.00$: superadditive (exploitable). \checkmark

Exact gap: $0.99 \times 0.01 \times 100 \times 100 \times 1000 / (1100 \times 1200) = 0.75$.
\end{example}
